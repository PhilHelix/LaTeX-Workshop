    \end{frame}
}

    \begin{frame}[c]
        \frametitle{Das steht heute auf dem Plan}
        \tableofcontents[
            currentsection,
            currentsubsection,
            subsectionstyle=show/shaded/hide,
            % hide subsubsection
            subsubsectionstyle=hide,
        ]
    \end{frame}
}
